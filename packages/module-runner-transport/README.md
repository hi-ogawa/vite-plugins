# @hiogawa/vite-plugin-module-runner-transport

User land plugin of https://github.com/vitejs/vite/pull/18485

## Usage

```js
// [vite.config.ts]
import { defineConfig } from "vite";
import moduleRunnerTransport from "@hiogawa/vite-plugin-module-runner-transport"

export default defineConfig({
  plugins: [
    moduleRunnerTransport(),
  ],
});
```
